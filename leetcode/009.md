  class Solution {
      public boolean isPalindrome(int x) {
          // 如果是负数就直接返回假
          if (x < 0) {
              return false;
          }
          int left = getNum(x);
          int right = 1;
          // 如果是个位数就返回真
          if (left == right) {
              return true;
          }
          // 对于多位数的处理
          while (left > right) {
              int weight_left = (int) Math.pow(10, left);
              int weight_left2 = (int) Math.pow(10, left-1);
              int weight_right = (int) Math.pow(10, right);
              int weight_right2 = (int) Math.pow(10, right-1);
  
              if ((x % weight_left) / weight_left2 != (x % weight_right)/weight_right2) {
                  return false;
              }
              left--;
              right++;
          }
          return true;
      }
  
      int getNum(int num) {
          int index = 0;
          while(num!=0){
              num/=10;
              index++;
          }
          return index;
      }
  }
  ```

      public boolean isPalindrome(int x) {
          if( x < 0){
              return false;
          }
          ArrayList<Integer> list = new ArrayList<Integer>();
          while(x >= 1){
              int temp = x % 10;
              x /= 10;
              list.add(temp);
          }
  
          for(int i=0, j=list.size()-1; i < j; ++i, --j){
              if(list.get(i) != list.get(j)){
                  return false;
              }
          }
          return true;
      }
  }

  class Solution {
      public boolean isPalindrome(int x) {
          // 特殊情况：
          // 如上所述，当 x < 0 时，x 不是回文数。
          // 同样地，如果数字的最后一位是 0，为了使该数字为回文，
          // 则其第一位数字也应该是 0
          // 只有 0 满足这一属性
          if (x < 0 || (x % 10 == 0 && x != 0)) {
              return false;
          }
  
          int revertedNumber = 0;
          while (x > revertedNumber) {
              revertedNumber = revertedNumber * 10 + x % 10;
              x /= 10;
          }
  
          // 当数字长度为奇数时，我们可以通过 revertedNumber/10 去除处于中位的数字。
          // 例如，当输入为 12321 时，在 while 循环的末尾我们可以得到 x = 12，revertedNumber = 123，
          // 由于处于中位的数字不影响回文（它总是与自己相等），所以我们可以简单地将其去除。
          return x == revertedNumber || x == revertedNumber / 10;
      }
  }
  ```
