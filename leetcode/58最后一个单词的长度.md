class Solution {
    public int lengthOfLastWord(String s) {
        s = s.trim();
        char[] temps = s.toCharArray();
        int max = 0;
        for (int i = temps.length - 1; i >= 0; i--) {
            if (temps[i] != ' ') {
                max += 1;
            } else {
                break;
            }
        }
        return max;
    }
}
