class Solution {
    public int[] searchRange(int[] nums, int target) {
        int left = bindarySearchLeft(nums, target);
        int right = bindarySearchRight(nums, target);
        if(left > right){
            return new int[]{-1, -1};
        }
        return new int[]{left, right};
    }
    public int bindarySearchLeft(int[] nums, int key) {
        int left = 0;
        int right = nums.length - 1;
        while (left <= right) {
            int mid = left + (right - left) / 2;
            if (nums[mid] >= key) {
                right -= 1;
            } else {
                left += 1;
            }
        }
        return left;
    }

    public int bindarySearchRight(int[] nums, int key) {
        int left = 0;
        int right = nums.length - 1;
        while (left <= right) {
            int mid = left + (right - left) / 2;
            if (nums[mid] <= key) {
                left += 1;
            } else {
                right -= 1;
            }
        }
        return right;
    }
}
