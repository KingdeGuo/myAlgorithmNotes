class Solution {
    public String longestCommonPrefix(String[] strs) {
        String ans = "";
        if (strs.length == 0) {
            return ans;
        }
        if (strs.length == 1) {
            return strs[0];
        }
        Arrays.sort(strs);
        StringBuilder stringBuilder = new StringBuilder();
        for (int i = 0; i < strs[0].length(); i++) {
            char temp1 = strs[0].charAt(i);
            char temp2 = strs[strs.length - 1].charAt(i);
            if (temp1 == temp2) {
                stringBuilder.append(temp1);
            } else {
                break;
            } 
        }

        return stringBuilder.toString();

    }
}
