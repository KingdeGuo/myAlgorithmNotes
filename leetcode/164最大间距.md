class Solution {
    public int maximumGap(int[] nums) {
        int length = nums.length;
        if (length == 0 || length == 1) {
            return 0;
        }
        if (length == 2) {
            return Math.abs(nums[0] - nums[1]);
        }
        Arrays.sort(nums);
        int max = nums[1] - nums[0];
        for (int i = 1; i < length - 1; i++) {
            max = Math.max(max, nums[i + 1] - nums[i]);
        }
        return max;
    }
}
