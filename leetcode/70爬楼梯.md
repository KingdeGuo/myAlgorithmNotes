class Solution {
    public int climbStairs(int n) {
        if (n == 0 || n == 1) {
            return 1;
        }
        
        int left = 1;
        int right = 1;
        int resu = 0;
        for (int i = 2; i < n+1; i++) {
            resu = left + right;
            left = right;
            right = resu;
        }

        return resu;
    }
    
}
