  import java.util.*;
  
  class Solution {
      public int romanToInt(String s) {
          int sum = 0;
          int preNum = getValue(s.charAt(0));
          for(int i = 1;i < s.length(); i ++) {
              int num = getValue(s.charAt(i));
              if(preNum < num) {
                  sum -= preNum;
              } else {
                  sum += preNum;
              }
              preNum = num;
          }
          sum += preNum;
          return sum;
      }
      
      private int getValue(char ch) {
          switch(ch) {
              case 'I': return 1;
              case 'V': return 5;
              case 'X': return 10;
              case 'L': return 50;
              case 'C': return 100;
              case 'D': return 500;
              case 'M': return 1000;
              default: return 0;
          }
      }
  }

  class Solution {
      public int romanToInt(String s) {
          s = s.replaceAll("IV","A");
          s = s.replaceAll("IX","B");
          s = s.replaceAll("XL","G");
          s = s.replaceAll("XC","H");
          s = s.replaceAll("CD","J");
          s = s.replaceAll("CM","K");
          char[] chars = s.toCharArray();
          int sum = 0;
          for(int i=0; i<chars.length; ++i){
              sum += getValue(chars[i]);
          }
          return sum;
      }
  
      int getValue(char c){
