class Solution {
    public int fib(int n) {
        if (n <= 1) {
            return n;
        }
        int p = 0;
        int q = 1;
        int ans = 0;
        for (int i = 1; i <= n; i++) {
            ans = (p + q) % 1000000007;
            q = p;
            p = ans;
        }
        return ans;
    }
}
