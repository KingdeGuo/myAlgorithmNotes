  ```java
  class Solution {
      public int reverse(int x) {
          boolean flag = true;
          if (x < 0) {
              flag = false;
          }
          String temp = Math.abs(x) + "";
          temp = new StringBuffer(temp).reverse().toString();
          if (flag == false) {
              temp = "-" + temp;
          }
          try {
              return (int) Integer.parseInt(temp);
          } catch (Exception e) {
              return 0;
          }
      }
  }
  ```

  ```java
  class Solution {
      public int reverse(int x) {
  	int ans = 0;
  	while (x != 0) {
  		if ((ans * 10) / 10 != ans) {
  			ans = 0;
  			break;
  		}
  		ans = ans * 10 + x % 10;
  		x = x / 10;
  	}
  	return ans;
      }
  }
  ```

  要注意防止溢出

  ```java
  class Solution {
      public int reverse(int x) {
          int rev = 0;
          while (x != 0) {
              int pop = x % 10;
              x /= 10;
              if (rev > Integer.MAX_VALUE/10 || (rev == Integer.MAX_VALUE / 10 && pop > 7)) return 0;
              if (rev < Integer.MIN_VALUE/10 || (rev == Integer.MIN_VALUE / 10 && pop < -8)) return 0;
              rev = rev * 10 + pop;
          }
          return rev;
      }
  }
  ```
