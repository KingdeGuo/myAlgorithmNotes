class Solution {
    public boolean isPowerOfTwo(int n) {
        if(n < 0){
            return false;
        }
        int tmp1 = n;
        int count = 0;
        while( n >= 1){
            n = n >> 1;
            count++;
        }
        count-=1;
        int tmp2 = 1 << count;
        return tmp1 == tmp2;
    }
}

class Solution {
    public boolean isPowerOfTwo(int n) {

        if (n <= 0) {
            return false;
        }
        
        String nums = Integer.toBinaryString(n);
        if (nums.charAt(0) != '1') {
            return false;
        }
        if (nums.length() >= 1) {
            for (int i = 1; i < nums.length(); i++) {
                if (nums.charAt(i) != '0') {
                    return false;
                }
            }
        }
        return true;
    }
}
