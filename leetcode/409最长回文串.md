class Solution {
    public int longestPalindrome(String s) {
        if(s.length() == 0){
            return 0;
        }
        HashSet<Character> set = new HashSet<>();
        char[] chars = s.toCharArray();
        int count = 0;
        for (int i = 0; i < chars.length; i++) {
            if (set.contains(chars[i])) {
                set.remove(chars[i]);
                count++;
            } else {
                set.add(chars[i]);
            } 
        }
        return set.isEmpty() ? count * 2 : count * 2 + 1;
    }
}
