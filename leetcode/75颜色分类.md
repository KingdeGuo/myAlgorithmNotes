class Solution {
    public void sortColors(int[] nums) {
        int[] colors = new int[3];
        int length = nums.length;
        // 如果长度是0或者1那就不用排序直接返回就可以了
        if (length == 0 || length == 1) {
            return;
        }
        // 如果长度是2
        if (length == 2) {
            // 如果顺序不对就换一下顺序
            if (nums[0] > nums[1]) {
                int temp;
                temp = nums[0];
                nums[0] = nums[1];
                nums[1] = temp;
            } else {
                // 如果顺序对了就直接返回
                return ;
            }
        }
        // 对于三个以上的就采用桶排序思路扫描
        for (int i = 0; i < colors.length; i++) {
            colors[i] = 0;
        }
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] == 0) {
                colors[0]++;
            } else if (nums[i] == 1) {
                colors[1]++;
            } else {
                colors[2]++;
            }
        }
        for (int i = 0; i < colors[0]; i++) {
            nums[i] = 0;
        }
        for (int i = colors[0]; i <colors[0] +  colors[1]; ++i) {
            nums[i] = 1;
        }
        for (int i = colors[0] +  colors[1]; i < colors[0] +  colors[1] + colors[2]; ++i) {
            nums[i] = 2;
        }

    }
}
